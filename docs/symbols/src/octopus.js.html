 16 (function() {	
 17 	var root = this;
 18 	var oldOctopus = root.Octopus;
 19 	
 20 	
 21 	/*
 22 	 * @namespace Octopus
 23 	 * @namespace _8 shortcut to octopus
 24 	 */
 25 	
 26 	var Octopus, _8;
 27 	if (typeof exports !== 'undefined') {
 28 		Octopus = _8 = exports
 29 	} else {
 30 		Octopus = root._8 = root.Octopus = {};
 31 	}
 32 	
 33 	
 34 	Octopus.VERSION = "0.1";
 35 
 36 	Octopus._traits = {};
 37 
 38 	// create a new tentacle (trait)
 39 	/*
 40 	 * @function t
 41 	 */
 42 	
 43 	Octopus.t =  function(name, proto) {
 44 		if(!name) throw "Octopus.t Invalid name"
 45 		if(!proto) return this._traits[id];
 46 		var construct = proto.construct;
 47 		delete proto.construct;
 48 		this._traits[name] = new OctoTrait(proto, construct);		
 49 	}
 50 	
 51 	// check to see if a trait has been added
 52 	Octopus.hasT = function(name) {
 53 		return (this._traits[name]);
 54 	}
 55 	Octopus._paths = {};
 56 	
 57 	// create object blueprint
 58 	Octopus.spawn = function(traits, proto) {
 59 		// create the a new base class each time so we can alter the prototype at will
 60 
 61 		if(!proto) proto = {};
 62 	 	
 63 		var fn = this._getBaseObj(proto.construct);
 64 	
 65 		_.each( getTraitList(traits), function(id) {
 66 			var t;
 67 			if(t = Octopus._traits[id]) {
 68 				fn.prototype._traits.push(id);
 69 				if(t.init) fn.prototype._inits.push(t.init);
 70 				_.extend(fn.prototype, t.proto);
 71 			}  
 72 		});
 73 		delete proto.construct;
 74 		_.extend(fn.prototype, proto);
 75 
 76 		return fn;
 77 	}
 78 	Octopus._getBaseObj = function(construct) {
 79 		var fn = function () {
 80 			 var self = this;
 81 	        _.each(this._inits, function(init) {
 82 				init.apply(self);
 83 	        });
 84 	
 85 			// fire the object constructor after all tentacles have been initialized
 86 	        this._useConstructor.apply(this, arguments);
 87 		}
 88 		fn.prototype._traits = [];
 89 		fn.prototype._inits = new Array();
 90 		fn.prototype._useConstructor = (_.isFunction(construct)) ? construct : function() {};
 91 		fn.prototype.hasT = function(name) {
 92 			return _.indexOf(this._traits, name) != -1;
 93 		}
 94 		return fn;
 95 	}
 96 	
 97 	// a package system
 98 	Octopus.path = function(p, addon) {
 99 		if(!_.isString(p)) throw "Octopus space path must be a String";
100 		return  Octopus._getPathObj(Octopus._paths, p.split("."), addon);
101 	}
102 	
103 	Octopus._getPathObj = function(parent, parts, addon) {
104 		
105 		var name = parts.shift();
106 		var	current = parent[name];
107 
108 		// create a new package if the one we are looking for is not there
109 		if (!current) 
110 			parent[name] = current = {};
111 			
112 		// continue traversing if we are not at the bottom
113 		if(parts.length) 
114 			return Octopus._getPathObj(current, parts, addon);
115 			
116 		// at the last path extend if needed
117 		if (addon) 
118 			_.extend(current, addon);
119 			
120 		return current;
121 	}
122 	
123 	Octopus.noConflict =  function() {
124 		root.Octopus = root._8 = oldOctopus;
125 		return Octopus;
126 	}
127 	
128 	// extend the Octopus namespace
129 	Octopus.learn = function(ext) {
130 		_.each(_.functions(ext), function(name){
131 			Octopus[name] = ext[name];
132 	    });
133 	};
134 	
135 	// run a anynomous function
136 	Octopus.go = function(imports, action) {
137 		var scope = (function() {});
138 		_.each(imports, function(p, name) {
139 			scope[name] = Octopus.path(p);
140 		});
141 		if(_.isFunction(action)) action.apply(scope, Array.prototype.slice.call(arguments, 2))
142 	}
143 	
144 	// a little fly weight obj to hold traits
145 	var OctoTrait = function(proto, init) {
146 		this.proto = (!proto) ? {} : proto;
147 		this.init = init;
148 	}
149 	
150 	var getTraitList = function(traits) {
151 		if(_.isArray(traits)) return traits;
152 		if(_.isString(traits)) return _.map(traits.split(","), function(t) {return trim(t)});
153 		return [];
154 	} 
155 	
156 	var trim = function(str) {
157 		return str.replace(/^\s*([\S\s]*?)\s*$/, '$1');
158 	}
159 
160 })();
